#include <stdio.h>

typedef struct date
{

}date;



void display(date d)

{
    
printf("The date is %d/%d/%d\n",d.date,d.month,d.year);

}




int datecmp(date d1, date d2)

{
    
//makes the decision on the comparisoin of year
    
if(d1.year > d2.year)
    
{
        
 return 1;
    
}
    
if(d1.year < d2.year)
    
{
        
 return -1;
    
}
    
//makes the decision on the comparisoin of month

{
        
 return 1;

 return -1;

{
        
 return 1;
    
}
    
if(d1.date < d2.date)
    
{
        
 return -1;
    
}
    
//returns o if the dates are same
    
return 0;
    

}
