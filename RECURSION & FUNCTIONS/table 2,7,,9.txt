//array of size 3*10 containing multiplication tables of 2,7 & 9 respectively



#include<stdio.h>

void printtable(int *multable, int num, int n)

{
    
printf("The multiplication table of %d follows..\n",num);
    
for(int i=0; i<n; i++)
    
{
        
 multable[i]= num *(i+1);
    
}
    

for(int i=0; i<n; i++)
    
{
        
 printf("%d x %d = %d\n",num,i,multable[i]);
    
}
    
printf("\n-----------------\n");
}



int main()

{
    
int multable[3][10];
    
printtable(multable[0],2,10);
    
printtable(multable[1],7,10);
    
printtable(multable[2],9,10);
    
return 0;

}
